% importing the package
\usepackage[framemethod=TikZ]{mdframed}

% setting counters
\newcounter{theocount}[section]\setcounter{theocount}{0}
\renewcommand{\thetheo}{\arabic{section}.\arabic{theocount}}

\newenvironment{theo}[2][]{
	% code for counter
	\refstepcounter{theocount}

	% box design code
	\ifstrempty{#1}
	% if no title
	{
		\mdfsetup{
			frametitle={
				\tikz[baseline=(current bonding box.east),outer sep=0pt]
				\node[anchor=east,rectangle,fill=blue!10]
				{\strut Theorem~\thetheo};
			}
		}
	}
	% if title
	{
		\mdfsetup{
			frametitle={
				\tikz[baseline=(current bonding box.east),outer sep=0pt]
				\node[anchor=east,rectangle,fill=blue!10]
				{\strut Theorem~\thetheo:~#1};
			}
		}
	}
	% defaults
	\mdfsetup{
		innertopmargin=10pt,
		linecolor=blue!20,
		linewidth=2pt,
		topline=true,
		frametitleaboveskip=\dimexpr-\ht\strutbox\relax
	}

	% code for environment name
	\begin{mdframed}[]\relax\label{#2}}
	{\end{mdframed}
}
